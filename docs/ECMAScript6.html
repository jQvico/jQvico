New feuteres in JavaScript (ECMAScript 6)

The following are some new enhancements of Java Script which are implemented in top web browsers.

Class

You can use class statement with the constructor method and the possibility of inheritance. 

    class boo extends foo{
        constructor(){
            this.a = 5;
        }
    }

Arrow functions

Functions with short syntax which resolve the problem of calling this statement inner them. 

    class foo{
        boo(){
            this.x = 1;
            $("a").click( () => { console.log(this.x) }) // 1
        }
    }

String templates

Pass and read a variable in the string. 

    var x = 5;
    var str = `Count: ${x} * ${x} = ${x*x}`;
    console.log(str) // Count: 5 * 5 = 25

Spread Operator

Spreading of an iterable element on single elements. 

    var x = [ 1, "c", false];
    var y = [ "a", "b", ...x ];
    console.log(y) //["a", "b", 1, "c", false]

Rest Parameter

Aggregate parameters into array. 

    function foo (x, y, ...z) {
        console.log(z);
    }
    foo("a", "b", 1, "c", false) //  [1, "c", false]

Variable types: let, const

let - scope within closest {} braces.
const - constant, not modifiable in the code, except object properties and array elements. 

    var i = 0;
    const x = 5;
    const y = {a: 1};
    for (let i = 1; i < 5; i++ ){
      console.log(i); // 1, 2, 3, 4
    }
    y.a = 4;
    console.log(i); // 0
    console.log(y); //  Object { a=4 }
    x = 10; // error

Function default parameaters

Pass default parameters into the function. 

    function foo(x=1, y=2){
        console.log(x, y);
    }
    foo(5); // 5 2

for of steatment

The statement looping over values of iterable objects. 

  var x = [1, 2, 3];
  for (let v of x) {
    console.log(v); //1, 2, 3
  }

Promises

Helpful in asynchronous calls. You can define what to do after execution of request in case of success and failure. 

  new Promise(function(resolve, reject) {
    setTimeout(function() { resolve('ok'); }, 1000);
  })
  .then(function(result) {
    console.log(result);  // 'ok'
  });

Generators

Generators are special functions allow control flow by using keyword yield. 

  function* gen() {
    yield 1;
    yield 2;
    yield 3;
  }

  var g = gen();
  console.log(g.next().value);  // 1
  console.log(g.next().value);  // 2

Map/Set

Methods lets you store values. 

  let s = new Set()
  s.add("Apple").add(3);
  s.has(3)

  let input = [0, 1, 2];
  let add = input.map((x)=>{return x + 1});
  console.log(add)  // 1, 2, 3

New Methods


  // Searching strings
  "javascript".includes("sc")  // true
  "javascript".startsWith("ava", 1)  // true

  // Truncate a floating point number.
  console.log(Math.trunc(10.8)) // 10

  // Assigning new properties to the object.
  var main  = { a: 0 }
  var ext = { b: 1, c: 'z' }

  Object.assign(main, ext);
  console.log(main) //  Object { a=0,  b=1,  c="z"}

  // Finding an element in the array.
  [ 1, 2, 'a', 'b' ].find(x => x == 'a') // a

  // Number type checking
  Number.isNaN(NaN) === true        // true
  Number.isFinite(NaN) === false    // true

  // String repeating
  "js-".repeat(3) // "js-js-js-"
